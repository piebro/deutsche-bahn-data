df = duckdb.sql("""
SELECT
    strftime(time, '%Y-%m') as months,
    AVG(delay_in_min) as "Durchschnittliche Verspätung (min)",
    COUNT(*) as "Anzahl Stopps",
    SUM(CASE WHEN is_canceled THEN 1 ELSE 0 END) as "Ausgefallene Stopps"
FROM '../monthly_processed_data/*.parquet'
GROUP BY months
ORDER BY months
""").df()

util.show_figure(
    [
        util.FigureConfig(
            title="Monatliche Durchschnittsverspätung",
            label="Durchschnittsverspätung",
            x_col="months",
            y_col="Durchschnittliche Verspätung (min)",
            query_or_df=df,
        )
    ]
)
